#include <WiFi.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>

const float latitude = 53.5303;   // Тольятти
const float longitude = 49.3461;

const char* ssid = "TP-Link_00E0";
const char* password = "68860097";

void connectWiFi() {
  WiFi.begin(ssid, password);
  Serial.print("Подключение к WiFi");

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("\nWiFi подключен!");
}

void getWeather() {
  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;

    String url = "https://api.open-meteo.com/v1/forecast?";
    url += "latitude=" + String(latitude, 4);
    url += "&longitude=" + String(longitude, 4);
    url += "&current_weather=true";
    url += "&hourly=relativehumidity_2m,precipitation,cloudcover,windspeed_10m,winddirection_10m";
    url += "&timezone=Europe/Moscow";

    http.begin(url);
    http.setTimeout(10000);
    int httpCode = http.GET();

    if (httpCode > 0) {
      Serial.printf("HTTP код: %d\n", httpCode);
      
      if (httpCode == HTTP_CODE_OK) {
        String payload = http.getString();
        
        DynamicJsonDocument doc(16384);
        DeserializationError error = deserializeJson(doc, payload);
        
        if (error) {
          Serial.print("Ошибка парсинга JSON: ");
          Serial.println(error.c_str());
          http.end();
          return;
        }
        
        Serial.println("\n ПОГОДНЫЕ ДАННЫЕ:");
        
        JsonObject current = doc["current_weather"];
        if (!current.isNull()) {
          float temp = current["temperature"];
          float windSpeed = current["windspeed"];
          float windDir = current["winddirection"];
          
          Serial.print("Температура: ");
          Serial.print(temp); Serial.println(" C");
          
          Serial.print("Скорость ветра: ");
          Serial.print(windSpeed); Serial.println(" км/ч");
          
          Serial.print("Направление ветра: ");
          Serial.print(windDir); Serial.println("°");
        }
        
        JsonObject hourly = doc["hourly"];
        if (!hourly.isNull()) {
          float humidity = hourly["relativehumidity_2m"][0];
          float precip = hourly["precipitation"][0];
          float cloud = hourly["cloudcover"][0];
          
          Serial.print("Влажность:");
          Serial.print(humidity); Serial.println(" %");
          
          Serial.print("Осадки: ");
          Serial.print(precip); Serial.println(" мм");
          
          Serial.print("Облачность: "); 
          Serial.print(cloud); Serial.println(" %");
        }
        
        float lat = doc["latitude"];
        float lon = doc["longitude"];
        float elev = doc["elevation"];
      }
    } else {
      Serial.print("Ошибка HTTP запроса: ");
      Serial.println(http.errorToString(httpCode).c_str());
    }
    
    http.end();
  } else {
    Serial.println("Нет подключения к WiFi");
  }
}

void setup() {
  Serial.begin(115200);
  delay(1000);
  
  Serial.println("\n\n=== ESP32-C3 ПОГОДНАЯ СТАНЦИЯ ===");
  connectWiFi();
}

void loop() {
  getWeather();
  delay(60 * 1000);
}
